<body>
  <div class="wrapper">
    <h2>My To-Do List</h2>
    <div class="inputs">
      <input type="text" id="taskText" placeholder="Task details">
      <input type="datetime-local" id="taskDate">
      <button onclick="addTask()">Add</button>
    </div>
    <ul id="list"></ul>
  </div>

  <script>
    function addTask() {
      const text = document.getElementById('taskText').value;
      const date = document.getElementById('taskDate').value;
      if (text.trim() === "") return;

      const li = document.createElement('li');
      li.innerHTML = `
        <span>${text} ${date ? `<br><small>⏰ ${date}</small>` : ''}</span>
        <div class="btns">
          <button class="done" onclick="toggleDone(this)">Done</button>
          <button class="edit" onclick="editTask(this)">Edit</button>
          <button class="delete" onclick="deleteTask(this)">Del</button>
        </div>
      `;

      document.getElementById('list').appendChild(li);
      document.getElementById('taskText').value = "";
      document.getElementById('taskDate').value = "";
    }

    function toggleDone(btn) {
      btn.closest('li').classList.toggle('done');
    }

    function editTask(btn) {
      const li = btn.closest('li');
      const span = li.querySelector('span');
      const current = span.innerText.split("⏰")[0].trim();
      const updated = prompt("Edit task:", current);
      if (updated !== null && updated.trim() !== "") {
        span.innerHTML = updated;
      }
    }

    function deleteTask(btn) {
      btn.closest('li').remove();
    }
  </script>
